public class Solution {
    public bool IsPerfectSquare(int num) {
        long low = 1;
        long high = num;
        while(low <= high) {
            long mid = low+(high-low)/2;
            long mid2 = mid*mid;
            if(mid2 == num) return true;
            else if(mid2 > num) high = mid-1;
            else low = mid+1;
        }
        return false;
    }
}
