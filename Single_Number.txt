public class Solution {
    public int SingleNumber(int[] nums) {
        if(nums.Length < 2) return nums[0];
        int sum = 0;
        var map = new Dictionary<int,int>();
        for(int i=0; i<nums.Length; i++) {
            if(map.ContainsKey(nums[i])) sum -= nums[i];
            else {
                map.Add(nums[i],i);
                sum += nums[i];
            }
        }
        return sum;
    }
}
