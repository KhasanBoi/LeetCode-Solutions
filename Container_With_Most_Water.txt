public class Solution {
    public int MaxArea(int[] height) {
        int max = int.MinValue;
        int left = 0;
        int right = height.Length-1;
        while(left < right) {
            int y = right-left;
            int x = Math.Min(height[right], height[left]);
            max = Math.Max(y*x, max);
            if(height[left] < height[right]) left++;
            else right--;
        }
        return max;
    }
}
