public class Solution {
    public int CountPrimes(int n) {
        bool[] primes = new bool[n];
        for(int i=2; i*i<primes.Length; i++) {
            if(!primes[i]) {
                for(int j=i; j*i<primes.Length; j++) {
                    primes[j*i] = true;
                }
            }
        }
        int count = 0;
        for(int i=2; i<primes.Length; i++) {
            if(!primes[i]) count++;
        }
        return count;
    }
}
