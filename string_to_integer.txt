public class Solution {
    public int MyAtoi(string s) {
        int res = 0;
		int sign = 1;
		int x = s.IndexOf('-'), y = s.IndexOf('+');
		if(x > y) sign = -1;
		for(int i=x+1; i<s.Length; i++) {
			if(Char.IsDigit(s[i])) res = res*10 + ((int)(s[i])-'0');
		}
		return res*sign;
    }
}
